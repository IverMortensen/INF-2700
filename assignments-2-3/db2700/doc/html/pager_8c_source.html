284static int put_header_int_at(page_p p, int offset, int val) {
285 if (offset < 0 || offset >= PAGE_HEADER_SIZE - INT_SIZE) {
286 put_msg(ERROR,

358static pqueue_p release_pqueue(pqueue_p q) {
359 if (!q) return 0;
360 pq_elm_p nextp = q->first;
361 pq_elm_p p;
362 while (nextp) {
363 p = nextp; nextp = p->next;
364 free(p);
365 if (nextp == q->last) {
366 if (nextp != q->first)
367 free(nextp); /* beware: already freed if last == first */
368 nextp = 0;
369 }
370 }
371 free(q);
372 q = 0;
373 return 0;
374}

383/* pg becomes the last in q */
384static void pq_insert(pqueue_p q, pq_elm_p p) {
385 if (!q->first) {
386 p->prev = p;
387 p->next = p;
388 q->first = p;
389 q->last = p;
390 } else {
391 p->prev = q->last;
392 p->next = q->first;
393 q->first->prev = p;
394 q->last->next = p;
395 q->last = p;
396 }
397 q->len++;
398 return;
399}
400
401static void pq_remove(pqueue_p q, pq_elm_p p) {
402 if (q->len <= 1) { /* at most one element in q */
403 q->first = 0;
404 q->last = 0;
405 q->len = 0;
406 } else {
407 p->next->prev = p->prev;
408 p->prev->next = p->next;
409 if (p == q->first)
410 q->first = p->next;
411 if (p == q->last)
412 q->last = p->prev;
413 q->len--;
414 }
415 return;
416}

426/* pg is moved to the last in the LRF queue */
427static void pq_touch(page_p pg) {
428 if (!pg) {
429 put_msg(WARN, "touching NULL page.\n");
430 return;
431 }
432 pq_elm_p p = pg->qelm;
433 if (!p) {
434 put_msg(WARN, "touching NULL qelm.\n");
435 return;
436 }
437
438 pqueue_p q = pg->pinned ? q_pinned : q_unpinned;
439 if (p == q->last) return;
440 pq_remove(q, p);
441 pq_insert(q, p);
442 return;
443}

445/* remove the first in q */
446static page_p pq_dequeue_unpinned() {
447 if (!q_unpinned->first) return 0;
448
449 pq_elm_p p = q_unpinned->first;
450 page_p pg = p->page;
451 pq_remove(q_unpinned, p);
452 free(p);
453 return pg;
454}

456static page_p pq_dequeue_pinned() {
457 if (!q_pinned->first) return 0;
458
459 pq_elm_p p = q_pinned->first;
460 page_p pg = p->page;
461 unpin(pg);
462 pq_remove(q_unpinned, p);
463 free(p);
464 return pg;

513static fhandle_p make_fhandle(char const* fname, int fd) {
514 fhandle_p fh = malloc(sizeof (file_handle_struct));
515 fh->fname = malloc(strlen(fname) + 1);
516 strcpy(fh->fname, fname);

534static fhandle_p open_tbl_file(char const* fname) {
535 if (num_file_handles == MAX_OPEN_FILES) {
536 put_msg(WARN,
537 "open Cannot file %s because the limit %d of open files has reached.",
538 fname, MAX_OPEN_FILES);
539 return 0;
540 }
541
542 int fd = open(fname, O_RDWR, 0);
543 if (fd == -1) {
544 /* if the file does not exist, create one */
545 if ((fd = creat(fname, 0600)) == -1) {
546 put_msg(WARN, "Failed to create file %s.", fname);
547 return 0;
548 }
549
550 /* close and open the created file again for read and write */
551 if (close(fd) == -1 || (fd = open(fname, O_RDWR, 0)) == -1)
552 return 0;
553 }
554
555 int empty_i = get_empty_fhandle_i();
556 if (empty_i == -1) return 0;
557
558 fhandle_p fh = make_fhandle(fname, fd);
559
560 file_handles[empty_i] = fh;
561 num_file_handles++;
562
563 return fh;
564}

602int pager_init(void) {
603 num_file_handles = 0;
604
605 /* global vars file_handles[] are initialized with NULL.
606 in case they are not, or pager_init is called again
607 after pager_terminate, initialize them anyway */
608 for (size_t i = 0; i < MAX_OPEN_FILES; i++)
609 file_handles[i] = 0;
610 for (size_t i = 0; i < NUM_PAGES; i++) {
611 pages[i] = make_page(i);
612 if (pages[i] == NULL) {
613 put_msg(ERROR, "pager_init failed");
614 pager_terminate();
615 return 0;
616 }
617 }
618 q_pinned = make_pqueue();
619 q_unpinned = make_pqueue();
620 pager_profiler_reset();
621 return 1;
622}

669static void release_block(block_p b) {
670 if (!b) return;
671 if (b->page->pinned)
672 unpin(b->page);
673 remove_blk_from_fhandle(b);
674 if (b->fhandle->current_block == b)
675 b->fhandle->current_block = 0;
676 b->page->block = 0;
677 free(b);

680void pager_terminate(void) {
681 /* put_pqueues_info (DEBUG); */
682 for (size_t i = 0; i < NUM_PAGES; i++) {
683 if (!pages[i]) continue;
684 release_block(pages[i]->block);
685 if (pages[i]) {
686 free(pages[i]->content);
687 free(pages[i]);
688 }
689 pages[i] = 0;
690 }
691 for (size_t i = 0; i < MAX_OPEN_FILES; i++)
692 close_tbl_file(file_handles[i]);
693 q_unpinned = release_pqueue(q_unpinned);
694 q_pinned = release_pqueue(q_pinned);
695}

697/* Find an available buffer page, in this order:
698 - unused page,
699 - unpinned page,
700 - unpin a pinned page.
701*/
702static page_p available_page() {
703 /* put_pqueues_info (DEBUG); */
704 page_p pg;
705 size_t i;
706 /* First, get an unused page */
707 if (q_pinned->len + q_unpinned->len < NUM_PAGES) {
708 for (i = 0; pages[i]->block; i++);
709 pg = pages[i];
710 } else {
711 /* put_msg (DEBUG, "available_page: all pages are used.\n"); */
712 pg = pq_dequeue_unpinned(); /* replace an unpinned page */
713 if (!pg)
714 /* put_msg(DEBUG, "available_page: all pages are pinned.\n"); */
715 /* If all pages are pinned, unpin the first page
716 and get that one.
717 This is of course a bad page replacement strategy. */
718 pg = pq_dequeue_pinned();
719
720 release_block(pg->block);
721 init_page(pg);
722 }
723 pq_enqueue(q_unpinned, pg);
724 return pg;
725}

727/* Returns the buffer page (position in pages[]) of the block).
728 If the block does not have a buffer page allocated, allocate one.
729*/
730static page_p page_for_block(block_p b) {
731 if (!b) return 0;
732 for (size_t i = 0; i < NUM_PAGES; i++) {
733 if (pages[i] && same_block(pages[i]->block, b))
734 return pages[i];
735 }
736 /* put_msg(WARN, "block %d not in mem yet, allocate an available one.\n", b->blk_nr); */
737 return available_page();
738}

740page_p get_page(char const* fname, int blknr) {
741 block_p blk = 0;
742 fhandle_p fh = get_tbl_file(fname);
743 if (!fh) fh = open_tbl_file(fname);
744
745 if (!fh) {
746 put_msg(ERROR, "get_page: NULL fh.\n");
747 return 0;
748 }
749
750 if (blknr == -1)
751 blknr = fh->num_blocks == 0 ? 0 : fh->num_blocks - 1;
752
753 if (blknr < 0 || blknr > fh->num_blocks) {
754 put_msg(ERROR, "get_page: block nr %d out of ragne [0,%d].",
755 blknr, fh->num_blocks);
756 return 0;
757 }
758
759 if (fh->current_block && blknr == fh->current_block->blk_nr)
760 blk = fh->current_block;
761 else if (blknr == fh->num_blocks)
762 fh->num_blocks++;
763 else
764 blk = get_buffered_blk_in_fhandle(fh, blknr);
765
766 if (!blk) {
767 blk = malloc(sizeof (block_struct));
768 blk->fhandle = fh;
769 blk->blk_nr = blknr;
770 if (pin(blk) == NULL) {
771 remove_blk_from_fhandle(blk);
772 free(blk);
773 return 0;
774 }
775 set_blk_in_fhandle(fh, blk);
776 blk->page->current_pos = PAGE_HEADER_SIZE;
777 }
778 /* put_msg (DEBUG, "get_page: blk %d, page %d\n",
779 blk->blk_nr, blk->page->page_nr); */
780 fh->current_block = blk;
781 return blk->page;
782}

824int read_page(page_p p) {
825 if (!p) {
826 put_msg(ERROR, "read_page: NULL page.\n");
827 return 0;
828 }
829 if (p->dirty) return 1;
830 if (!p->block) {
831 put_msg(ERROR, "read_page: NULL block.\n");
832 return 0;
833 }
834 if (!p->block->fhandle) {
835 put_msg(ERROR, "read_page: NULL fhandle.\n");
836 return 0;
837 }
838 int fd = p->block->fhandle->fd;
839 if (lseek(fd, (off_t) BLOCK_SIZE * p->block->blk_nr, SEEK_SET) < 0) {
840 put_msg(ERROR, "read_page: lseek to fd %d offset %ld fails.\n",
841 fd, BLOCK_SIZE * p->block->blk_nr);
842 return 0;
843 }
844 int bytes_read = read(fd, p->content, BLOCK_SIZE);
845 if (bytes_read == -1) return 0;
846 if (bytes_read == 0)
847 set_page_free_pos(p, PAGE_HEADER_SIZE);
848 else {
849 inc_num_reads(fd, p->block->blk_nr);
850 check_page_header_size(p);
851 set_page_free_pos_from_content(p);
852 }
853 return 1;
854}

856int write_page(page_p p) {
857 if (!p) {
858 put_msg(ERROR, "write_page: NULL page.\n");
859 return 0;
860 }
861 if (!p->dirty) return 1;
862 if (!p->block) return 0;
863 if (!p->block->fhandle) return 0;
864
865 int fd = p->block->fhandle->fd;
866
867 if (lseek(fd, (off_t) BLOCK_SIZE * p->block->blk_nr, SEEK_SET) < 0)
868 return 0;
869 inc_num_writes(fd, p->block->blk_nr);
870 p->dirty = 0;
871 if (write(fd, p->content, BLOCK_SIZE) == -1) return 0;
872 return 1;
873}
