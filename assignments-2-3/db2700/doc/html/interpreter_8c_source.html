37static int init_with_options (int argc, char* argv[]) {
38 char cmd_file[MAX_LINE_WIDTH] = "";
39 char db_dir[MAX_LINE_WIDTH] = "";
40 int c;
41
42 msglevel = INFO;
43
44 while ((c = getopt(argc, argv, "hm:d:c:")) != -1)
45 switch (c) {
46 case 'h':
47 printf("Usage: runtest [switches]\n");
48 printf("\t-h help, print this message\n");
49 printf("\t-m [fewid] msg level [fatal,error,warn,info,debug]\n");
50 printf("\t-d db_dir default to ./tests/testfront\n");
51 printf("\t-c cmd file eg. ./tests/testcmd.dbcmd, default to stdin\n");
52 exit(0);
53 case 'm':
54 switch (optarg[0]) {
55 case 'f': msglevel = FATAL; break;
56 case 'e': msglevel = ERROR; break;
57 case 'w': msglevel = WARN; break;
58 case 'i': msglevel = INFO; break;
59 case 'd': msglevel = DEBUG; break;
60 }
61 break;
62 case 'd':
63 strcpy(db_dir, optarg);
64 case 'c':
65 strcpy(cmd_file, optarg);
66 break;
67 case '?':
68 if (optopt == 'm' || optopt == 'd' || optopt == 'c')
69 printf("Option -%c requires an argument.\n", optopt);
70 else if (isprint(optopt))
71 printf("Unknown option `-%c'.\n", optopt);
72 else
73 printf("Unknown option character `\\x%x'.\n", optopt);
74 abort();
75 default:
76 abort();
77 }
78
79 if (cmd_file[0] == '\0')
80 in_s = stdin;
81 else {
82 struct stat in_stat;
83 stat(cmd_file, &in_stat);
84 if (S_ISREG(in_stat.st_mode)) {
85 in_s = fopen( cmd_file, "r");
86 if (in_s == NULL) {
87 printf("Cannot open file: %s\n", cmd_file);
88 in_s = stdin;
89 } else
90 put_msg(DEBUG, "file \"%s\" is open for read.\n", cmd_file);
91 } else {
92 printf("\n\"%s\" is not a regular file, fall back to stdin ...\n\n",
93 cmd_file);
94 in_s = stdin;
95 }
96 }
97
98 if (in_s == stdin) {
99 printf("Welcome to db2700 session\n");
100 printf(" - Enter \"help\" for instructions\n");
101 printf(" - Enter \"quit\" to leave the session\n");
102 }
103
104 if (db_dir[0] == '\0')
105 strcpy(db_dir, "./tests/testfront");
106
107 if (!set_system_dir(db_dir)) {
108 put_msg(ERROR, "cannot set database at %s\n", db_dir);
109 return 0;
110 }
111
112 return open_db();
113}

116 char format[8];

136/* Used to read till the end of a statement or a parenthesis.
137 For simplicity, we require that a statement
138 - does not go beyond a single line
139 - ends with a semicolon
140 - does not contain a semicolon */
141static int read_till(char* stmnt_str, char c) {
142 char format[16];
143 snprintf(format, sizeof format, "%%%u[^%c\n];", MAX_LINE_WIDTH, c);
144 if (fscanf(in_s, format, stmnt_str))

163/* split str into substrings, separated by char c,
164 which is not white space.
165 A substring allows exactly the given number of white spaces
166 (only 0 or 1 for our usage). */
167static int str_split(char* str, char c, char* substrs[],
168 int max_count, int n_white_space) {
169 if ((n_white_space != 0) && (n_white_space != 1)) {
170 put_msg(ERROR,
171 "str_split: only 0 or 1 white space in substrings allowed.\n");
172 return 0;
173 }
174
175 size_t count = 0;
176 char *p;
177
178 for (p = str; *p != '\0'; p++) {
179 if (*p == c) {
180 *p = '\0';
181 if (++count > max_count) {
182 put_msg(DEBUG, "str_split: too many substrings.\n");
183 return 0;
184 }
185 }
186 }
187 count++;
188
189 char str_tmp1[MAX_TOKEN_LEN] = "",
190 str_tmp2[MAX_TOKEN_LEN] = "",
191 str_tmp3[MAX_TOKEN_LEN] = "";
192 p = str;
193
194 for (int i = 0; i < count; i++, p += strlen(p) + 1) {
195 int n_tmps = sscanf(p,"%s%s%s", str_tmp1, str_tmp2, str_tmp3);
196
197 if (n_tmps != n_white_space + 1) {
198 put_msg(DEBUG,
199 "str_split: empty string or too many white spaces.\n");
200 release_strs(substrs, i);
201 return 0;
202 }
203 if (n_white_space == 0) {
204 substrs[i] = malloc(strlen(str_tmp1) + 1);
205 strcpy(substrs[i], str_tmp1);
206 } else {
207 substrs[i] = malloc(strlen(str_tmp1) + strlen(str_tmp2) + 2);
208 strcpy(substrs[i], str_tmp1);
209 strcat(substrs[i], " ");
210 strcat(substrs[i], str_tmp2);
211 }
212 }
213
214 return count;
215}

270static void create_tbl() {
271 char tbl_name[MAX_TOKEN_LEN], token[MAX_TOKEN_LEN];
272
273 if (!next_token(token)) {
274 put_msg(ERROR, "Must create something.\n");
275 return;
276 }
277 if (strcmp(token, t_table) != 0) {
278 put_msg(ERROR, "Cannot create \"%s\".\n", token);
279 return;
280 }
281 if (!next_token(tbl_name)) {
282 put_msg(ERROR, "create table: missing table name.\n");
283 return;
284 }
285 if (next_char() != '(') {
286 error_near(0);
287 return;
288 }
289
290 put_msg(DEBUG, "create table name: \"%s\".\n", tbl_name);
291
292 char attrs_str[MAX_LINE_WIDTH];
293
294 if (!read_till(attrs_str, ')')) {
295 error_near(0);
296 return;
297 }
298 if (next_char() != ')' && next_char() != ';') {
299 error_near(0);
300 skip_line();
301 return;
302 }
303 skip_line();
304
305 schema_p sch = get_schema(tbl_name);
306
307 if (sch) {
308 put_msg(ERROR, "Table \"%s\" already exists.\n", tbl_name);
309 return;
310 }
311
312 char* attrs[MAX_ATTRS];
313 int num_attrs = str_split(attrs_str, ',', attrs, MAX_ATTRS, 1);
314 if (num_attrs == 0) {
315 put_msg(ERROR, "create table %s: incorrect attributes\n", tbl_name);
316 return;
317 }
318
319 sch = new_schema(tbl_name);
320
321 char attr_name[MAX_TOKEN_LEN], attr_type[MAX_TOKEN_LEN];
322 int str_attr_len;
323
324 for (int i = 0; i < num_attrs; i++) {
325 if (sscanf(attrs[i], "%s%s", attr_name, attr_type) != 2) {
326 put_msg(ERROR, "create table %s: incorrect attribute \"%s\"\n",
327 tbl_name, attrs[i]);
328 goto abort_create;
329 }
330
331 if (strcmp(attr_type, t_int) == 0) {
332 add_field(sch, new_int_field(attr_name));
333 } else if (sscanf(attr_type, "str[%d]", &str_attr_len) == 1) {
334 add_field(sch, new_str_field(attr_name, str_attr_len));
335 } else {
336 put_msg(ERROR, "create table %s: unknown type \"%s\" for attribute \"%s\"\n",
337 tbl_name, attr_type, attr_name);
338 goto abort_create;
339 }
340 }
341
342 release_strs(attrs, num_attrs);
343 return;
344
345 abort_create:
346 release_strs(attrs, num_attrs);
347 remove_schema(sch);
348}

350static void drop_tbl() {
351 char tbl_name[MAX_TOKEN_LEN], token[MAX_TOKEN_LEN];
352
353 if (!next_token(token) || strcmp(token, t_table) != 0) {
354 put_msg(ERROR, "drop what?\n");
355 skip_line();
356 return;
357 }
358 if (!next_token(tbl_name) || tbl_name[0] == '#') {
359 put_msg(ERROR, "drop table: nothing to drop.\n");
360 skip_line();
361 return;
362 }
363
364 char *p = strchr(tbl_name, ';');
365 if (p) {
366 *p = 0;
367 } else {
368 if (next_char() != ';') {
369 put_msg(ERROR, "drop table: syntax error (missing ';').\n");
370 skip_line();
371 return;
372 }
373 }
374
375 skip_line();
376
377 put_msg(DEBUG, "drop table name: \"%s\".\n", tbl_name);
378 remove_table(get_table(tbl_name));
379}

407 char tbl_name[MAX_TOKEN_LEN], token[MAX_TOKEN_LEN];
408
409 if (!next_token(token) || strcmp(token, t_into)) {

418 put_msg(ERROR, "\"insert into %s\" must be followed with \"values\".\n",
419 tbl_name);
420 return;

422 if (next_char() != '(') {
423 error_near(0);
424 return;
425 }
426
427 put_msg(DEBUG, "insert into: \"%s\".\n", tbl_name);
428 schema_p sch = get_schema(tbl_name);
429 if (!sch) {
430 put_msg(ERROR, "Schema \"%s\" does not exist.\n", tbl_name);
431 skip_line();
432 return;
433 }
434
435 char vals_str[MAX_LINE_WIDTH];
436
437 if (!read_till(vals_str, ')')) {
438 error_near(0);
439 return;
440 }
441 if (next_char() != ')' && next_char() != ';') {
442 error_near(0);
443 skip_line();
444 return;
445 }
446 skip_line();
447
448 char* vals[MAX_ATTRS];
449 int num_vals = str_split(vals_str, ',', vals, schema_num_flds(sch), 0);
450 if (num_vals == 0) {

452 return;
453 }
454
455 /* put_schema_info(DEBUG, sch); */
456
457 record rec = new_filled_record(sch, vals);
458 release_strs(vals, num_vals);
459
460 if (rec) {
461 append_record(rec, sch);
462 release_record(rec, sch);
463 }
464}

494static select_desc* parse_select() {
495 select_desc *slct = new_select_desc();
496 char in_str[MAX_LINE_WIDTH] = "";
497 char from_str[MAX_TOKEN_LEN] = "", join_with[MAX_TOKEN_LEN] = "";
498 char *where_str, *join_str, *p;
499
500 if (!read_till(in_str, ';')) {
501 error_near("select ");
502 return 0;
503 }
504
505 p = strstr(in_str, " from ");
506 if (!p) {
507 put_msg(ERROR, "select %s: from which table to select?\n", in_str);
508 release_select_desc(slct);
509 return 0;
510 }
511 *p = '\0';
512 p += 6;
513 if (sscanf(p, "%s", from_str) != 1) {
514 put_msg(ERROR, "select from what?\n");
515 release_select_desc(slct);
516 return 0;
517 }
518 slct->from_tbl = get_table(from_str);
519 if (!slct->from_tbl) {
520 put_msg(ERROR, "select: table \"%s\" does not exist.\n", from_str);
521 release_select_desc(slct);
522 return 0;
523 }
524
525 slct->num_attrs = str_split(in_str, ',', slct->attrs, MAX_ATTRS, 0);
526 if (slct->num_attrs == 0) {
527 put_msg(ERROR, "select from %s: select what?\n", p);
528 release_select_desc(slct);
529 return 0;
530 }
531
532 p += strlen(from_str);
533 join_str = strstr(p, " natural join ");
534 if (join_str) {
535 join_str += 14;
536 put_msg(DEBUG, "from: \"%s\", natural join: \"%s\"\n",
537 from_str, join_str);
538 if (sscanf(join_str, "%s", join_with) != 1) {
539 put_msg(ERROR, "natural join with \"%s\" is not supported.\n",
540 join_str);
541 release_select_desc(slct);
542 return 0;
543 }
544 if (strcmp(from_str, join_str) == 0) {
545 put_msg(ERROR, "natural join on same table is not supported.\n");
546 release_select_desc(slct);
547 return 0;
548 }
549 slct->right_tbl = get_table(join_with);
550 if (!slct->right_tbl) {
551 put_msg(ERROR, "natural join: table \"%s\" does not exist.\n",
552 join_with);
553 release_select_desc(slct);
554 return 0;
555 }
556 }
557
558 where_str = strstr(p, " where ");
559 if (where_str) where_str += 7;
560
561 put_msg(DEBUG, "from: \"%s\", where: \"%s\"\n", from_str, where_str);
562
563 if (where_str) {
564 if (sscanf(where_str, "%s %s %d",
565 slct->where_attr, slct->where_op, &slct->where_val)
566 != 3) {
567 put_msg(ERROR, "query \"%s\" is not supported.\n", where_str);
568 release_select_desc(slct);
569 return 0;
570 }
571 }
572 return slct;
573}

575static void select_rows() {
576 select_desc *slct = parse_select();
577 tbl_p where_tbl = 0, res_tbl = 0, join_tbl = 0;
578
579 if (!slct) return;
580
581 if (slct->right_tbl) {
582 join_tbl = table_natural_join(slct->from_tbl, slct->right_tbl);
583 if (!join_tbl) {
584 release_select_desc(slct);
585 return;
586 }
587 }
588
589 if (slct->where_attr[0] != '\0' && slct->where_op[0] != '\0') {
590 where_tbl = table_search(join_tbl ? join_tbl : slct->from_tbl,
591 slct->where_attr,
592 slct->where_op,
593 slct->where_val);
594 if (!where_tbl) {
595 release_select_desc(slct);
596 return;
597 }
598 }
599
600 if (slct->attrs[0][0] == '*')
601 table_display(where_tbl ? where_tbl
602 : (join_tbl ? join_tbl : slct->from_tbl));
603 else {
604 res_tbl = table_project(where_tbl ? where_tbl : slct->from_tbl,
605 slct->num_attrs, slct->attrs);
606 table_display(res_tbl);
607 }
608
609 remove_table(join_tbl);
610 remove_table(where_tbl);
611 remove_table(res_tbl);
612 release_select_desc(slct);
613}

615void interpret(int argc, char* argv[]) {
616 if (!init_with_options(argc, argv))
617 exit(EXIT_FAILURE);
618
619 char token[MAX_TOKEN_LEN];
620
621 while (!feof(in_s)) {
622 if (in_s == stdin)
623 printf("db2700> ");
624 if (!next_token(token)) {
625 put_msg(ERROR, "Getting input failed\n");
626 exit(EXIT_FAILURE);
627 }
628 put_msg(DEBUG, "current token is \"%s\".\n", token);
629 if (strcmp(token, t_quit) == 0) { quit(); break;}
630 if (token[0] == '#')
631 { skip_line(); continue; }
632 if (strcmp(token, t_help) == 0)
633 { show_help_info(); continue; }
634 if (strcmp(token, t_show) == 0)
635 { show_database(); continue; }
636 if (strcmp(token, t_print) == 0)
637 { print_str(); continue; }
638 if (strcmp(token, t_create) == 0)
639 { create_tbl(); continue; }
640 if (strcmp(token, t_drop) == 0)
641 { drop_tbl(); continue; }
642 if (strcmp(token, t_insert) == 0)
643 { insert_row(); continue; }
644 if (strcmp(token, t_select) == 0)
645 { select_rows(); continue; }
646 error_near(token);
647 }
648}
