248static void save_tbl_descs() {
249 /* backup the descriptors first in case we need some manual investigation */
250 char *tbl_desc_backup = concat_names("__backup", "_", tables_desc_file);
251 rename(tables_desc_file, tbl_desc_backup);
252 free(tbl_desc_backup);
253
254 FILE *dbfile = fopen(tables_desc_file, "w");
255 tbl_p tbl = db_tables, next_tbl = 0;
256 while (tbl) {
257 save_tbl_desc(dbfile, tbl);
258 release_schema(tbl->sch);
259 next_tbl = tbl->next;
260 free(tbl);
261 tbl = next_tbl;
262 }
263 fclose(dbfile);
264}

266static void read_tbl_descs() {
267 FILE *fp = fopen(tables_desc_file, "r");
268 if (!fp) return;
269 char name[30] = "";
270 schema_p sch = NULL;
271 field_desc_p fld = NULL;
272 int num_flds = 0, fld_type, fld_len;
273 while (!feof(fp)) {
274 if (fscanf(fp, "%s %d\n", name, &num_flds) < 2) {
275 fclose(fp);
276 return;
277 }
278 sch = new_schema(name);
279 for (size_t i = 0; i < num_flds; i++) {
280 fscanf(fp, "%s %d %d", name, &(fld_type), &(fld_len));
281 switch (fld_type) {
282 case INT_TYPE:
283 fld = new_int_field(name);
284 break;
285 case STR_TYPE:
286 fld = new_str_field(name, fld_len);
287 break;
288 }
289 fscanf(fp, "%d\n", &(fld->offset));
290 add_field(sch, fld);
291 }
292 fscanf(fp, "%d\n", &(sch->tbl->num_records));
293 }
294 db_tables = sch->tbl;
295 fclose(fp);
296}

395static schema_p make_sub_schema(schema_p s, int num_fields, char *fields[]) {
396 if (!s) return 0;

399 schema_p res = new_schema(sub_sch_name);
400 free(sub_sch_name);

402 field_desc_p f = 0;

417int add_field(schema_p s, field_desc_p f) {
418 if (!s) return 0;
419 if (s->len + f->len > BLOCK_SIZE - PAGE_HEADER_SIZE) {
420 put_msg(ERROR,
421 "schema already has %d bytes, adding %d will exceed limited %d bytes.\n",
422 s->len, f->len, BLOCK_SIZE - PAGE_HEADER_SIZE);
423 return 0;
424 }
425 if (s->num_fields == 0) {
426 s->first = f;
427 f->offset = 0;
428 }
429 else {
430 s->last->next = f;
431 f->offset = s->len;
432 }
433 s->last = f;
434 s->num_fields++;
435 s->len += f->len;
436 return s->num_fields;
437}

558static page_p get_page_for_next_record(schema_p s) {
559 page_p pg = s->tbl->current_pg;
560 if (peof(pg)) return 0;
561 if (eop(pg)) {
562 unpin(pg);
563 pg = get_next_page(pg);
564 if (!pg) {
565 put_msg(FATAL, "get_page_for_next_record failed at block %d\n",
566 page_block_nr(pg) + 1);
567 exit(EXIT_FAILURE);
568 }
569 page_set_pos_begin(pg);
570 s->tbl->current_pg = pg;
571 }
572 return pg;
573}

575static int get_page_record(page_p p, record r, schema_p s) {
576 if (!p) return 0;
577 if (!page_valid_pos_for_get_with_schema(p, s)) {
578 put_msg(FATAL, "try to get record at invalid position.\n");
579 exit(EXIT_FAILURE);
580 }
581 field_desc_p fld_desc;
582 size_t i = 0;
583 for (fld_desc = s->first; fld_desc;
584 fld_desc = fld_desc->next, i++)
585 if (is_int_field(fld_desc))
586 assign_int_field(r[i], page_get_int(p));
587 else
588 page_get_str(p, r[i], fld_desc->len);
589 return 1;
590}

616static int find_record_int_val(record r, schema_p s, int offset,
617 int (*op) (int, int), int val) {
618 page_p pg = get_page_for_next_record(s);
619 if (!pg) return 0;
620 int pos, rec_val;
621 for (; pg; pg = get_page_for_next_record(s)) {
622 pos = page_current_pos(pg);
623 rec_val = page_get_int_at (pg, pos + offset);
624 if ((*op) (val, rec_val)) {
625 page_set_current_pos(pg, pos);
626 get_page_record(pg, r, s);
627 return 1;
628 }
629 else
630 page_set_current_pos(pg, pos + s->len);
631 }
632 return 0;
633}

667void append_record(record r, schema_p s) {
668 tbl_p tbl = s->tbl;
669 page_p pg = get_page_for_append(s->name);
670 if (!pg) {
671 put_msg(FATAL, "Failed to get page for appending to \"%s\".\n",
672 s->name);
673 exit(EXIT_FAILURE);
674 }
675 if (!put_page_record(pg, r, s)) {
676 /* not enough space in the current page */
677 unpin(pg);
678 pg = get_next_page(pg);
679 if (!pg) {
680 put_msg(FATAL, "Failed to get page for \"%s\" block %d.\n",
681 s->name, page_block_nr(pg) + 1);
682 exit(EXIT_FAILURE);
683 }
684 if (!put_page_record(pg, r, s)) {
685 put_msg(FATAL, "Failed to put record to page for \"%s\" block %d.\n",
686 s->name, page_block_nr(pg) + 1);
687 exit(EXIT_FAILURE);
688 }
689 }
690 tbl->current_pg = pg;
691 tbl->num_records++;
692}

739tbl_p table_search(tbl_p t, char const* attr, char const* op, int val) {
740 if (!t) return 0;
741
742 int (*cmp_op)() = 0;
743
744 if (strcmp(op, "=") == 0)
745 cmp_op = int_equal;
746
747 if (strcmp(op, "<") == 0)
